    import React from "react";
import {
  withTheme,
  ScreenContainer,
  Image,
  Button,
  IconButton
} from "@draftbit/ui";
import {StyleSheet, Text, View, ActivityIndicator, Alert, StatusBar} from "react-native";
import { Rating } from "react-native-ratings";
import { uploadVideo } from "../utils";
import Images from "../../config/Images";
import Routes from "../../config/Routes";
import { ContentPicker, ChangePasswordModal } from "../components";
import API from "../../config/api";

class ProfileScreen extends React.Component {
  state = {
    modalVisible: false,
    changePasswordModalVisible: false,
    uploadFeedback: false,
    profile: {}
  };

  componentDidMount() {
    this.fetchProfile();
  }

  fetchProfile = async () => {
    const res = await API.get("/detail");
    this.setState({profile: res.data});
    console.log(res.data);
  }

  upload = async content => {
    this.setState({ modalVisible: false });
    this.setState({ uploadFeedback: true });
    await uploadVideo(content);
    Alert.alert(
      "Success!",
      "Your video was successfully uploaded!",
      [{ text: "OK", onPress: () => console.log("ok pressed") }],
      { cancelable: false }
    );
    this.setState({ uploadFeedback: false });
  };

  logout = () => {
    const { navigation } = this.props;
    navigation.navigate(Routes.LoginScreen);
  };

  goToMyVideos = () => {
    const { navigation } = this.props;
    navigation.navigate(Routes.MyVideoList);
  };

  render() {

    const { theme } = this.props;
    const { navigation } = this.props;

    const { modalVisible, changePasswordModalVisible } = this.state;
    return (
      <ScreenContainer hasSafeArea style={styles.screen}>
        <IconButton
          style={styles.settings}
          icon="settings"
          color={theme.colors.primary}
          onPress={this.logout}
        />
        <Image source={Images.Logo} resizeMode="contain" style={styles.pic} />
        <Text style={styles.email}>{this.state.profile.email}</Text>
        <Button
          style={styles.button}
          type="solid"
          onPress={() => this.setState({ modalVisible: true })}
        >
          Upload Video
        </Button>


        <Button
          type="solid"
          onPress={this.goToMyVideos}
        >
          My Video List
        </Button>


        <Button
          style={styles.button}
          type="solid"
          onPress={() => this.setState({ changePasswordModalVisible: true })}
        >
          Change Password
        </Button>

        {this.state.uploadFeedback ? (
          <Text style={{ fontSize: 25, color: "#000", textAlign: "center" }}>
            {" "}
            Please wait while your video uploads!{" "}
          </Text>
        ) : null}
        <ActivityIndicator
          size="large"
          color="#0000ff"
          animating={this.state.uploadFeedback}
        />
        <View style={{flex: 1, alignSelf: "stretch"}}>
          <ProfileRating ratingName='Content' ratingValue={this.state.profile.content}/>
          <ProfileRating ratingName='Language' ratingValue={this.state.profile.lang}/>
          <ProfileRating ratingName='Visual' ratingValue={this.state.profile.visual}/>
          <ProfileRating ratingName='Clarity' ratingValue={this.state.profile.clarity}/>
          <ProfileRating ratingName='Brevity' ratingValue={this.state.profile.brevity}/>
        </View>
        <ContentPicker
          modalVisible={modalVisible}
          closeModal={() => this.setState({ modalVisible: false })}
          isVideo
          handleContent={content => this.upload(content)}
        />
        <ChangePasswordModal
          modalVisible={changePasswordModalVisible}
          closeModal={() =>
            this.setState({ changePasswordModalVisible: false })
          }
        />
      </ScreenContainer>
    );
  }
}

function ProfileRating(props) {
  const {ratingName, ratingValue} = props
  return (
    <View style={styles.ratingBox}>
      <Text style={styles.ratingText}>{ratingName}</Text>
      <Rating
        style={styles.rating}
        startingValue={ratingValue}
        imageSize={30}
        readonly={true}
      />
    </View>
  )
}

const styles = StyleSheet.create({
  settings: {
    position: "absolute",
    top: 10,
    right: 10
  },
  screen: {
    alignItems: "center"
  },
  email: {
    fontWeight: "bold",
    fontSize: 20
  },
  pic: {
    width: 100,
    height: 100
  },
  MainContainer: {
    // Setting up View inside content in Vertically center.
    justifyContent: "center",
    flex: 1,
    margin: 10
  },
  button: {
    margin: 4
  },
  ratingBox: {
    flex: 1,
    alignItems: "center",
    flexDirection: "row",
    justifyContent: "center"
  },
  ratingText: {
    flex: 1,
    fontSize: 16,
    textAlign: "right"
  },
  rating: {
    flex: 2
  }
});

export default withTheme(ProfileScreen);
