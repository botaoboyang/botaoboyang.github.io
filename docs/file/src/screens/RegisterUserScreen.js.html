    src/screens/RegisterUserScreen.js

    import React from "react";
import {
  withTheme,
  TextField,
  Image,
  Container,
  ScreenContainer,
  Button,
  Touchable
} from "@draftbit/ui";
import { StyleSheet, Text, KeyboardAvoidingView, Alert } from "react-native";

import Images from "../../config/Images";
import Routes from "../../config/Routes";
import API from "../../config/api";

class RegisterUserScreen extends React.Component {
  state = {
    emailInput: "",
    passwordInput: "",
    confirmPasswordInput: ""
  };

  registerUser = () => {
    const { navigation } = this.props;
    const { emailInput, passwordInput, confirmPasswordInput } = this.state;

    if (passwordInput === confirmPasswordInput) {
      // Create account
      API.post("/register", {
        email: emailInput,
        password: passwordInput
      }).then(
        res => {
          console.log(res.data);
          console.log(res.status);
          console.log(res.data.errorMsg);
          // If user error, by default .then needs to have status 200
          if (res.data.errorMsg) {
            Alert.alert(
              "Registration Error",
              res.data.errorMsg,
              [{ text: "OK", onPress: () => console.log("OK Pressed") }],
              { cancelable: false }
            );
            navigation.navigate(Routes.LoginScreen);
          }

          // status 200 and no error message meaning success
          else {
            Alert.alert(
              "Registration successfull",
              "You have successfully registered please log in",
              [{ text: "OK", onPress: () => console.log("OK Pressed") }],
              { cancelable: false }
            );
            navigation.navigate(Routes.LoginScreen);
          }
          // status not 200 something went wrong on server display as an uknown error
        },
        error => {
          console.log(error);
          Alert.alert(
            "Unknown Error",
            "Oops Something went wrong! Please try again!",
            [{ text: "OK", onPress: () => console.log("OK Pressed") }],
            { cancelable: false }
          );
          //navigation.navigate(Routes.RegisterUserScreen)
        }
      );
    } else {
      alert("Password fields must match! Please try again");
    }
  };

  goToLogin = () => {
    const { navigation } = this.props;
    navigation.navigate(Routes.LoginScreen);
  };

  render() {
    const { theme } = this.props;

    return (
      <ScreenContainer hasSafeArea={true} scrollable={true}>
        <KeyboardAvoidingView
          enabled={true}
          behavior="position"
          keyboardVerticalOffset={20}
          style={styles.keyboardAvoidingViewBW}
        >
          <Container
            elevation={0}
            useThemeGutterPadding={true}
            style={styles.containerZl}
          >
            <Image
              source={Images.Logo}
              resizeMode="contain"
              style={styles.image4O}
            />
            <Container elevation={0} useThemeGutterPadding={true}>
              <TextField
                onChangeText={emailInput => this.setState({ emailInput })}
                value={this.state.emailInput}
                type="underline"
                label="Email"
                keyboardType="email-address"
                autoCapitalize="none"
                style={StyleSheet.flatten([
                  styles.textFieldDg,
                  { borderColor: theme.colors.surface }
                ])}
              />
              <TextField
                secureTextEntry={true}
                onChangeText={passwordInput => this.setState({ passwordInput })}
                keyboardType="default"
                type="underline"
                label="Password"
                value={this.state.passwordInput}
                style={styles.textFieldAB}
              />
              <TextField
                onChangeText={confirmPasswordInput =>
                  this.setState({ confirmPasswordInput })
                }
                label="Confirm Password"
                value={this.state.confirmPasswordInput}
                type="underline"
                keyboardType="default"
                secureTextEntry={true}
                style={styles.textField9W}
              />
              <Button
                color={theme.colors.primary}
                type="solid"
                style={styles.buttonOo}
                onPress={this.registerUser}
              >
                Create Account
              </Button>
              <Touchable style={styles.touchableTu} onPress={this.goToLogin}>
                <Text
                  style={StyleSheet.flatten([
                    styles.textKj,
                    theme.typography.button,
                    { color: theme.colors.primary }
                  ])}
                >
                  Already Registered? Sign In
                </Text>
              </Touchable>
            </Container>
            <Container
              useThemeGutterPadding={true}
              elevation={0}
              style={StyleSheet.flatten([
                styles.containerFi,
                { backgroundColor: theme.colors.surface }
              ])}
            >
              <Text
                style={StyleSheet.flatten([
                  styles.textDo,
                  theme.typography.caption,
                  { color: theme.colors.light }
                ])}
              >
                By creating an account, you agree to our Terms of Service,
                Privacy Policy and Cookie Policy.{" "}
              </Text>
            </Container>
          </Container>
        </KeyboardAvoidingView>
      </ScreenContainer>
    );
  }
}

const styles = StyleSheet.create({
  textFieldDg: {
    height: 82
  },
  image4O: {
    height: 125,
    alignSelf: "center",
    width: 125,
    marginTop: 48
  },
  textDo: {
    textAlign: "center",
    width: "100%"
  },
  containerZl: {
    height: "100%",
    flexGrow: 1,
    justifyContent: "space-between"
  },
  keyboardAvoidingViewBW: {
    flexGrow: 1
  },
  textKj: {
    textAlign: "center",
    width: "100%"
  },
  textFieldAB: {
    height: 82
  },
  textField9W: {
    height: 82
  },
  buttonOo: {
    height: 48,
    marginBottom: 24,
    marginTop: 24
  },
  containerFi: {
    marginBottom: 24
  },
  touchableTu: {
    marginBottom: 16
  }
});

export default withTheme(RegisterUserScreen);
