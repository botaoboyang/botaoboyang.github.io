    src/screens/LoginScreen.js

    import React from "react";
import {
  withTheme,
  Image,
  Container,
  Touchable,
  ScreenContainer,
  TextField,
  Button
} from "@draftbit/ui";

import { StyleSheet, Text, KeyboardAvoidingView } from "react-native";
import Images from "../../config/Images";
import Routes from "../../config/Routes";
import API from "../../config/api";

export class EmailPasswordLogin extends React.Component {
  state = {
    emailInput: "",
    passwordInput: ""
  };

  login = async () => {
    const { navigation } = this.props;
    let { emailInput, passwordInput } = this.state;

    // This method is used for testing
    if (emailInput === "q") {
      emailInput = "q@q.com";
      passwordInput = "123456";
    }

    const res = await API.post("/login", {
      email: emailInput,
      password: passwordInput
    });
    if (res.data && res.data["access_token"]) {
      const auth = `Bearer ${res.data["access_token"]}`;
      API.defaults.headers.common["Authorization"] = auth;
      navigation.navigate(Routes.Main);
    } else {
      alert("Oops, something went wrong!");
    }
  };

  createAccountOnPress = () => {
    const { navigation } = this.props;
    navigation.navigate(Routes.RegisterUserScreen);
  };

  render() {
    const { theme } = this.props;

    return (
      <ScreenContainer scrollable hasSafeArea>
        <KeyboardAvoidingView
          enabled
          behavior="padding"
          keyboardVerticalOffset={0}
          style={styles.keyboardAvoidingView}
        >
          <Container
            elevation={0}
            useThemeGutterPadding
            style={styles.logoAppNameContainer}
          >
            <Image
              source={Images.Logo}
              resizeMode="contain"
              style={styles.logo}
            />
            <Text
              style={StyleSheet.flatten([
                styles.appNameText,
                theme.typography.headline5,
                { color: theme.colors.primary }
              ])}
            >
              Video App for Teaching (VAT)
            </Text>
          </Container>
          <Container elevation={0} useThemeGutterPadding>
            <Text
              style={StyleSheet.flatten([
                styles.enterCredentialsLabel,
                theme.typography.overline,
                { color: theme.colors.medium }
              ])}
            >
              ENTER YOUR CREDENTIALS:
            </Text>
            <TextField
              type="solid"
              label="Email Address"
              placeholder="joe@example.com"
              keyboardType="email-address"
              value={this.state.emailInput}
              onChangeText={emailInput => this.setState({ emailInput })}
              autoCapitalize="none"
              style={styles.emailInput}
            />
            <TextField
              secureTextEntry
              value={this.state.passwordInput}
              onChangeText={passwordInput => this.setState({ passwordInput })}
              label="Password"
              type="solid"
              placeholder="**********"
              style={styles.passwordInput}
            />
            <Button
              type="solid"
              color={theme.colors.primary}
              style={styles.signInButton}
              onPress={this.login}
            >
              SIGN IN
            </Button>
          </Container>
          <Container
            useThemeGutterPadding
            elevation={0}
            style={styles.bottomContainer}
          >
            <Touchable
              style={styles.createAccountTouchable}
              onPress={this.createAccountOnPress}
            >
              <Text
                style={StyleSheet.flatten([
                  styles.createAccountText,
                  theme.typography.button,
                  { color: theme.colors.primary }
                ])}
              >
                Create Account
              </Text>
            </Touchable>
            <Text
              style={StyleSheet.flatten([
                styles.legalText,
                theme.typography.caption,
                { color: theme.colors.light }
              ])}
            >
              By tapping "Sign In", you agree to our Terms of Service, Privacy
              Policy and Cookie Policy.{" "}
            </Text>
          </Container>
        </KeyboardAvoidingView>
      </ScreenContainer>
    );
  }
}

const styles = StyleSheet.create({
  keyboardAvoidingView: {
    flexGrow: 1,
    justifyContent: "space-around"
  },
  logoAppNameContainer: {
    alignItems: "center"
  },
  logo: {
    width: 100,
    height: 100
  },
  appNameText: {
    width: "100%",
    textAlign: "center"
  },
  enterCredentialsLabel: {
    marginBottom: 16,
    textAlign: "center",
    width: "100%"
  },
  emailInput: {
    height: 82,
    marginBottom: 8
  },
  passwordInput: {
    height: 82
  },
  signInButton: {
    height: 48,
    marginBottom: 24,
    marginTop: 24
  },
  bottomContainer: {
    justifyContent: "flex-end"
  },
  createAccountTouchable: {
    width: "100%"
  },
  createAccountText: {
    textAlign: "center"
  },
  legalText: {
    marginTop: 24,
    textAlign: "center"
  }
});

export default withTheme(EmailPasswordLogin);
