    src/components/ChangePasswordModal.js

    import React from "react";

import {
  Text,
  View,
  TouchableWithoutFeedback,
  StyleSheet,
  TextInput,
  Keyboard,
  Alert
} from "react-native";
import Modal from "react-native-modal";
import { withTheme, Container, TextField, TextArea } from "@draftbit/ui";

import PromptButton from "./PromptButton";
import API from "../../config/api";

class ChangePasswordModal extends React.Component {
  state = {
    currentPasswordInput: "",
    newPasswordInput: "",
    confirmPasswordInput: ""
  };

  sleep = ms => {
    return new Promise(resolve => setTimeout(resolve, ms));
  };

  changePassword = async () => {
    const {
      currentPasswordInput,
      newPasswordInput,
      confirmPasswordInput
    } = this.state;
    if (newPasswordInput === confirmPasswordInput) {
      await API.post("/changePass", {
        oldPass: currentPasswordInput,
        newPass: newPasswordInput
      });
      Alert.alert("Password successfully changed!");
      await this.sleep(2000)
      this.props.closeModal()
    } else {
      Alert.alert("Passwords do not match!");
    }
    this.setState({
      currentPasswordInput: "",
      newPasswordInput: "",
      confirmPasswordInput: ""
    });
  };

  render() {
    const { modalVisible, closeModal } = this.props;
    const {
      currentPasswordInput,
      newPasswordInput,
      confirmPasswordInput
    } = this.state;
    return (
      <Modal isVisible={modalVisible}>
        <TouchableWithoutFeedback onPress={Keyboard.dismiss}>
          <View style={styles.modalContent}>
            <Text style={styles.headline}>
              What would you like your new password to be?
            </Text>
            <TextField
              secureTextEntry
              type="solid"
              label="Current Password"
              placeholder="Current Password"
              keyboardType="default"
              value={currentPasswordInput}
              onChangeText={currentPasswordInput =>
                this.setState({ currentPasswordInput })
              }
              style={styles.newPasswordInput}
            />
            <TextField
              secureTextEntry
              type="solid"
              label="New Password"
              placeholder="New Password"
              keyboardType="default"
              value={newPasswordInput}
              onChangeText={newPasswordInput =>
                this.setState({ newPasswordInput })
              }
              style={styles.newPasswordInput}
            />
            <TextField
              secureTextEntry
              type="solid"
              label="Confirm Password"
              placeholder="Confirm Password"
              keyboardType="default"
              value={confirmPasswordInput}
              onChangeText={confirmPasswordInput =>
                this.setState({ confirmPasswordInput })
              }
              style={styles.newPasswordInput}
            />
            <PromptButton text="Submit" onPress={this.changePassword} />
            <PromptButton text="Close" onPress={closeModal} />
          </View>
        </TouchableWithoutFeedback>
      </Modal>
    );
  }
}

const styles = StyleSheet.create({
  container: {
    alignItems: "flex-end"
  },
  modalContent: {
    flexDirection: "column",
    backgroundColor: "rgba(251, 252, 253, 100)",
    padding: 22,
    justifyContent: "center",
    alignItems: "stretch",
    borderRadius: 4,
    borderColor: "rgba(90, 69, 255, 100)",
    borderWidth: 5
  },
  newPasswordInput: {
    height: 60,
    marginBottom: 16
  },
  picker: {
    borderColor: "red",
    borderWidth: 3
  },
  headline: {
    fontSize: 20,
    fontWeight: "bold",
    textAlign: "center",
    marginBottom: 16
  }
});

export default ChangePasswordModal;
