    import React from "react";
import RNPickerSelect from "react-native-picker-select";
import { Card } from "react-native-elements";
import {
  Text,
  View,
  TouchableOpacity,
  TouchableWithoutFeedback,
  StyleSheet,
  Picker,
  TextInput,
  Keyboard
} from "react-native";
import Modal from "react-native-modal";
import {
  withTheme,
  Image,
  Container,
  Touchable,
  ScreenContainer,
  TextField,
  TextArea,
  Button
} from "@draftbit/ui";

import {
  takePhotoWithCamera,
  getImageFromCameraRoll,
  takeVideoWithCamera,
  getVideoFromCameraRoll
} from "../utils";
import PromptButton from "./PromptButton"

class ContentPicker extends React.Component {
  state = {
    videoTitle: "",
    videoDiscription: "",
    access: ""
  };
  updateSelection = choice => {
    //console.log("exevuted update")
    //console.log(choice);

    this.setState({ access: choice });
    //console.log("ACCESS BELOW")
    //console.log(this.state.access)
  };

  static defaultProps = {
    handleContent: () => {},
    isVideo: false
  };

  handleContent = async content => {
    if (content && !content.cancelled) {
      //this.props.closeModal();
      content.videoTitle = this.state.videoTitle;
      content.videoDiscription = this.state.videoDiscription;
      content.access = this.state.access;
      //content.videoTitle = this.state.this.props.closeModal();
      await this.props.handleContent(content);
    }
  };

  render() {
    const { modalVisible, closeModal, isVideo } = this.props;
    const { placeholder } = "select your access";
    return (
      <Modal isVisible={modalVisible} style={styles.modalContent}>
        <TouchableWithoutFeedback onPress={Keyboard.dismiss}>
          <View>
            <Card
              containerStyle={styles.cardStyle}
              title="Please pick public for your video to be visible to everyone and private for it to be visible to yourself only"
              titleStyle={styles.title}
            >
              <Card>
                <RNPickerSelect
                  style={styles.picker}
                  placeholder={placeholder}
                  onValueChange={value => this.updateSelection(value)}
                  items={[
                    { label: "Private", value: "private" },
                    { label: "Public", value: "public" }
                  ]}
                />
              </Card>
            </Card>
            <TextField
              type="solid"
              label="Video Title"
              multiline={true}
              placeholder="Enter your video title"
              keyboardType="default"
              value={this.state.videoTitle}
              onChangeText={videoTitle => this.setState({ videoTitle })}
              style={styles.emailInput}
            />
            <TextField
              type="solid"
              multiline={true}
              numberOfLines={4}
              label="Video Description"
              placeholder="Enter your video description"
              keyboardType="default"
              value={this.state.videoDiscription}
              onChangeText={videoDiscription =>
                this.setState({ videoDiscription })
              }
              style={styles.emailInput}
            />

            <PromptButton
              text={isVideo ? "Take Video" : "Take Picture"}
              onPress={async () => {
                if (isVideo) {
                  const video = await takeVideoWithCamera();
                  await this.handleContent(video);
                  return;
                }
                const image = await takePhotoWithCamera(true);
                await this.handleContent(image);
              }}
            />
            <PromptButton
              text={
                isVideo
                  ? "Select Video From Library"
                  : "Select Image From Library"
              }
              onPress={async () => {
                if (isVideo) {
                  const video = await getVideoFromCameraRoll();
                  await this.handleContent(video);
                } else {
                  const image = await getImageFromCameraRoll(true);
                  await this.handleContent(image);
                }
              }}
            />
            <PromptButton text="Close" onPress={() => closeModal()} />
          </View>
        </TouchableWithoutFeedback>
      </Modal>
    );
  }
}

  }

export default withTheme(ContentPicker);
