    import { createAppContainer, createSwitchNavigator } from "react-navigation";
import { createStackNavigator } from "react-navigation-stack";
import { createBottomTabNavigator } from "react-navigation-tabs";
import * as Screens from "./screens";
import React, {Component} from "react";
import {submitNewTitle} from "./screens/VideoDetailScreen.js";
import {
  View,
  Dimensions,
  StyleSheet,
  Text,
} from "react-native";

const AppNavigator = createSwitchNavigator(
  {
    AuthenticationNavigator: createStackNavigator(
      {
        LoginScreen: {
          screen: Screens.LoginScreen,
          navigationOptions: () => ({ title: "Login" })
        },
        RegisterUserScreen: {
          screen: Screens.RegisterUserScreen,
          navigationOptions: () => ({ title: "Register User" })
        }
      },
      {}
    ),
    MainNavigator: createBottomTabNavigator(
      {
        Main: {
          screen: Screens.MainScreen,
          navigationOptions: () => ({ title: "Main" })
        },
        VideoListNavigator: {
          navigationOptions: () => ({ tabBarLabel: "Video List" }),
          screen: createStackNavigator(
            {
              VideoListScreen: {
                screen: Screens.VideoListScreen,
                navigationOptions: () => ({
                  headerShown: false,
                  title: "Video List",
                })
              },
              VideoDetailScreen: {
                screen: Screens.VideoDetailScreen,
                navigationOptions: ({ navigation }) => ({
                  title: navigation.getParam("video_title"),
                  //headerRight: <Text onPress= {() => submitNewTitle()}>[Edit]</Text>
                })
              }
            },
            { initialRouteName: "VideoListScreen" }
          )
        },
        // VideoListScreen: {
        //   screen: Screens.VideoListScreen,
        //   navigationOptions: () => ({ title: "Video List" })
        // },
        ProfileScreen: {
          screen: Screens.ProfileScreen,
          navigationOptions: () => ({ title: "Profile" })
        },
        MyVideoList: {
          screen: Screens.MyVideoList,
          navigationOptions: () => ({ title: "MyVideoList" })
        }
      },
      {}
    )
  },
  {
    initialRouteName: "AuthenticationNavigator"
  }
);

const AppContainer = createAppContainer(AppNavigator);

export default AppContainer;
