export const uploadVideo = async videoObj => {
  const { cancelled, uri } = videoObj;
  if (!cancelled) {
    const { key: aws_key, access: aws_access } = await storeFileInS3(uri);
    const variables = {
      aws_access,
      aws_key,
      title: videoObj.videoTitle,
      description: videoObj.videoDiscription,
      is_public: aws_access === videoObj.access
    };
    await API.post("/video/upload", { ...variables });
  }

};

export const getFileFromS3 = async (key, access = "public") => {
  return await Storage.get(key, { level: access });
};

export const storeFileInS3 = async (
  fileUri,
  awsKey = null,
  access = "public"
) => {
  const blob = await new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.onload = function() {
      resolve(xhr.response);
    };
    xhr.onerror = function() {
      reject(new TypeError("Network request failed"));
    };
    xhr.responseType = "blob";
    xhr.open("GET", fileUri, true);
    xhr.send(null);
  });
  const { name, type } = blob._data;
  const options = {
    level: access,
    contentType: type
  };
  const key = awsKey || name;
  const result = await Storage.put(key, blob, options);
  return {
    access,
    key: result.key
  };
};

export const getCameraPermission = async () => {
  const { status } = await ImagePicker.requestCameraPermissionsAsync();
  return status === "granted";
};

export const getCameraRollPermission = async () => {
  const { status } = await ImagePicker.requestCameraRollPermissionsAsync();
  return status === "granted";
};

export const getImage = async (fromCameraRoll = true) => {
  const { updateParentState } = this.props;

  let img;
  if (fromCameraRoll) {
    img = await getImageFromCameraRoll(true);
  } else {
    img = await takePhotoWithCamera(true);
  }

  if (img && !img.cancelled) {
    this.setState({ avatar: img, skipped: false });
    await updateParentState({ avatar: { uri: img.uri, skipped: false } });
  }
};

export const takePhotoWithCamera = async (square = false) => {
  const options = square
    ? {
        quality: 0.5,
        base64: true,
        mediaTypes: "Images",
        allowsEditing: true,
        exif: false,
        aspect: [1, 1]
      }
    : {
        quality: 0.5,
        base64: true,
        mediaTypes: "Images",
        allowsEditing: true,
        exif: false
      };
  const permission = await getCameraPermission();
  const permission2 = await getCameraRollPermission();
  if (permission && permission2) {
    try {
      const image = await ImagePicker.launchCameraAsync(options);
      return image;
    } catch (err) {
      return null;
    }
  } else {
    return null;
  }
};

export const getImageFromCameraRoll = async (square = false) => {
  const options = square
    ? {
        quality: 0.5,
        base64: true,
        mediaTypes: "Images",
        allowsEditing: true,
        exif: false,
        aspect: [1, 1]
      }
    : {
        quality: 0.5,
        base64: true,
        mediaTypes: "Images",
        allowsEditing: true,
        exif: false
      };
  const permission = await getCameraRollPermission();
  if (permission) {
    const image = await ImagePicker.launchImageLibraryAsync(options);
    return image;
  }
  return null;
};

const videoIsUnderTimeLimit = ({ duration }) => {
  const MAX_VIDEO_LENGTH_SECONDS = 600; // 10 minutes
  return duration <= MAX_VIDEO_LENGTH_SECONDS * 1000;
};

export const getVideoFromCameraRoll = async () => {
  const options = {
    quality: 0.5,
    base64: true,
    mediaTypes: "Videos",
    exif: false,
    allowsEditing: true
  };
  const permission = await getCameraRollPermission();
  if (permission) {
    const video = await ImagePicker.launchImageLibraryAsync(options);
    if (!videoIsUnderTimeLimit(video)) {
      return null;
    }
    return video;
  }
  return null;
};
