import pytest
from .test_login import register_random_user
from .conftest import create_auth_header

def test_success(client):

    email, password, auth_headers = instance_of_auth_header(client)
    body, res = post_change_pass(client, auth_headers, password, "test")
    assert res.json['msg'] == 'success'
    login_client(client, email, body['newPass'])

def test_authorization(client):
    body = {
        'oldPass': '123456',
        'newPass': 'test'
    }
    res = client.post("/changePass", json=body)
    assert res.status_code == 401

def test_wrong_password(client):

    email, password, auth_headers = instance_of_auth_header(client)
    body, res = post_change_pass(client, auth_headers, password + '1', "test")
    assert res.json['errorMsg'] == 'Wrong password'
    login_client(client, email, body['newPass'])

@pytest.mark.parametrize("passwrd_num", [1, 2, 3, 4, 5])
def test_multiple_change_password(client, passwrd_num):

    email, password, auth_headers = instance_of_auth_header(client)
    body, res = post_change_pass(client, auth_headers, password, f'<test{passwrd_num}>')
    assert res.json['msg'] == 'success'
    login_client(client, email, body['newPass'])


def instance_of_auth_header(client):
    email = 'test@illinois.edu'
    password = '123456'
    auth_headers = create_auth_header(client, email=email, password=password)
    return email, password, auth_headers


def post_change_pass(client, auth_headers, oldPass, newPass):
    body = {
        'oldPass' : oldPass,
        'newPass' : newPass
    }
    res = client.post("/changePass", json=body, headers=auth_headers)
    assert res.status_code == 200
    return body, res


def login_client(client, email, password):
    body = {
        'email' : email,
        'password' : password
    }
    res = client.post("/login", json=body)
    assert res.status_code == 200
    return body
